    public int Tur_Adeti(int id){
        try {
            statement = con.createStatement();
            String query = "select count(*) as count from netflix_programtur where program='" +id+ "'";
            ResultSet cnt = statement.executeQuery(query);
            int count =0;
            while(cnt.next()){
                count = cnt.getInt("count");
            }
            return count;
        } catch (SQLException ex) {
            Logger.getLogger(VeriTabaniIslemleri.class.getName()).log(Level.SEVERE, null, ex);
            return 0;
        }
    }
    
    public String Turler(int id){
        try {
            int count = Tur_Adeti(id);
            statement = con.createStatement();
            String query = "select t.tur_adi From netflix_turler t LEFT JOIN netflix_programtur pt ON pt.tur=t.tur_id where pt.program='"+id+"'";
            ResultSet row = statement.executeQuery(query);
            String Turler = "";
            int i = 0;
            while(row.next()){
                Turler += row.getString("tur_adi");
                if(i != count-1){
                    if(i == count-2){
                        Turler +=" ve ";
                    } else {
                        Turler +=", ";
                    }
                }
                i++;
            }
            return Turler;
        } catch (SQLException ex) {
            Logger.getLogger(VeriTabaniIslemleri.class.getName()).log(Level.SEVERE, null, ex);
            return null;
        }
    }
    
    public ArrayList<Programlar> ProgramlariGetir(){
        ArrayList<Programlar> cikti = new ArrayList<Programlar>();
        try {
            statement = con.createStatement();
            String sorgu = "Select * from netflix_programlar";
            ResultSet rs = statement.executeQuery(sorgu);
            while(rs.next()){
                int id = rs.getInt("program_id");
                String program_adi = rs.getString("program_adi");
                String tip = rs.getString("tip");
                String bolum = rs.getString("bolum");
                String uzunluk = rs.getString("uzunluk");
                String Turler = Turler(id);
                cikti.add(new Programlar(id, program_adi, tip, Turler, bolum, uzunluk));
            }
            return cikti;
        } catch (SQLException ex) {
            Logger.getLogger(VeriTabaniIslemleri.class.getName()).log(Level.SEVERE, null, ex);
            return null;
        }
    }
    
    public String[][] KayitProgramListesi(int tur){
        String[][] veriler = new String[2][5];        
        try {
            statement = con.createStatement();
            String sorgu = "SELECT p.* FROM `netflix_programlar` p "
                    + "LEFT JOIN netflix_programtur pt ON p.program_id=pt.program "
                    + "LEFT JOIN netflix_turler t ON t.tur_id = pt.tur "
                    + "LEFT JOIN netflix_kullaniciprogram kp ON p.program_id = kp.program WHERE pt.tur='"+tur+"' GROUP BY pt.program ORDER BY kp.puan DESC LIMIT 0,2";
            ResultSet rs = statement.executeQuery(sorgu);
            int i=0;
            while(rs.next()){
                int id = rs.getInt("program_id");
                String Turler = Turler(id);
		veriler[i][0] = rs.getString("program_adi");
                veriler[i][1] = rs.getString("tip");
                veriler[i][2] = Turler;
                veriler[i][3] = rs.getString("bolum");
                veriler[i][4] = rs.getString("uzunluk");
                i++;
            }
            return veriler;
        } catch (SQLException ex) {
            Logger.getLogger(VeriTabaniIslemleri.class.getName()).log(Level.SEVERE, null, ex);
            return null;
        }
    }

        } catch (SQLException ex) {
            Logger.getLogger(VeriTabaniIslemleri.class.getName()).log(Level.SEVERE, null, ex);
            return null;
        }
    }

    private void kayitol_buttonActionPerformed(java.awt.event.ActionEvent evt) {                                               
        mesaj_yazisi.setText("");
        String AdSoyad = adsoyad_alani.getText();
        String Eposta = eposta_alani.getText();
        String DogumTarihi = dogum_tarihi_alani.getText();
        String password = new String(sifre_alani.getPassword());
        
        if((("".equals(AdSoyad) || "".equals(Eposta)) || "".equals(DogumTarihi)) || "".equals(password)){
            mesaj_yazisi.setText("Lütfen alanları doldurunuz.");
        } else {
            boolean kayit_varmi = islemler.KullcaniciKontrol(Eposta);
            if(!kayit_varmi){
                islemler.KullaniciEkle(AdSoyad,Eposta,DogumTarihi,password);
                mesaj_yazisi.setText("Kaydınız başarılı bir şekilde yapıldı.");
            } else {
                mesaj_yazisi.setText("Girilen epostaya ait bir kullanıcı mevcuttur.");
            }
        }
    }

    public void ProgramlariGoruntule(){
        model.setRowCount(0); // tabloyu boşaltıyoruz
        int sayac = 1;
        for (Map.Entry<String,Integer> entry : liste.entrySet()) {
                int tur = entry.getValue();
                for (int i = 0; i < 2; i++) {
                    String[][] veriler = new String[2][5];
                    veriler = islemler.KayitProgramListesi(tur);
                    String program_adi = veriler[i][0];
                    String tip = veriler[i][1];
                    String turler = veriler[i][2];
                    String bolum = veriler[i][3];
                    String uzunluk = veriler[i][4] + " Dk";
                    Object[] ekle = {program_adi,tip,turler,bolum,uzunluk};
                    model.addRow(ekle);
                }
            if(sayac == 3) break;
            sayac++;
        }
    }

        this.tip = tip;
        this.tur = tur;
        this.bolum = bolum;
        this.uzunluk = uzunluk;
    }

    }

        return program_adi;
    }

        this.program_adi = program_adi;
    }

    public String getTip() {
        return tip;
    }

    public void setTip(String tip) {
        this.tip = tip;
    }

    public String getTur() {
        return tur;
    }

        this.tur = tur;
    }

    public String getBolum() {
        return bolum;
    }

    }

        icerikler = islemler.ProgramlariGetir();
        
        if(icerikler != null){

                model.addRow(ekle);
            }
        }
    }

    public void ProgramKontrol(int islem){
        boolean kayit = islemler.ProgramIzlemeKontrol(this.program_id, this.Kullanici);
        int user_id = islemler.UserIdBul(this.Kullanici);
        if(islem == 0){
            if(kayit){ // kayit var ise izlenme süresini ve bölümünü güncelliyoruz
                int bolum = 2;
                String sure = "00:00:00";
                islemler.KullaniciIzlemeGuncelle(this.program_id,user_id,bolum,sure);
            } else {// kayıt yok ise yeni izleme kaydı oluşturuyoruz
                islemler.KullaniciIzlemeOlustur(this.program_id,user_id);
            }
        } else {
            if(kayit){ // kayit var ise izlenme süresini ve bölümünü güncelliyoruz
                int cevap = JOptionPane.showConfirmDialog((Component)null, "Programa kaldığınız yerden devam etmek istiyor musunuz?", "Bilgilendirme Mesajı", JOptionPane.YES_NO_OPTION);
                //System.out.println("Cevap : " + cevap);
                int bolum;
                String sure;
                if(cevap == 0){ // Yes
                    bolum = 1;
                    sure = "00:34:23";
                } else { // No
                    bolum = 1;
                    sure = "00:00:00";
                }
                islemler.KullaniciIzlemeGuncelle(this.program_id,user_id,bolum,sure);
            }
        }
    }
